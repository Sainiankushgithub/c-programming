// SUM OF THE FINITE DIGIT NUMBER
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int digit;
    printf("Enter any finite digit\n");
    scanf("%d",&digit);
    int sum=0;
    while(digit>0)
    {
        int r;
        r=digit%10;
        sum=sum+r;
        digit=digit/10;
    }
    printf("The sum of any finie digit is %d",sum);
    return 0;
}
