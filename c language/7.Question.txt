// CHECK WHETHER THE NUMBER IS ARMSTRONG OR NOT????
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int num;
    int r,sum=0;
    // ARMSTRONG NUMBER LIKE---> 371, 407
    printf("Enter any number to check whether it is armstrong or not\n");
    scanf("%d",&num);
    int temp=num;
    while(num>0)
    {
        r=num%10;
        sum=sum+r*r*r;
        num=num/10;
    }
    if(temp==sum)
    {
        printf("Entered number %d is an Armstrog number\n",temp);
    }
    else{
         printf("Entered number %d is not an  Armstrog number\n",temp);
    }
    return 0;
}
