// FIND WHETHER THE YEAR IS LEAP YEAR ARE NOT
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    printf("Enter any year\n");
    scanf("%d",&n);
    if(n%4==0)
    {
        printf("Entered year %d is a leap year",n);
    }
    else 
    {
        printf("Entered year %d is not a leap year",n);
    }
    return 0;
}
