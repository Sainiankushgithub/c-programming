void array3(int size, int arr1[],int arr2[])
{
    int arr3[size];
    cout<<"Displaying the third array\n";
    for(int i=0;i<size;i++)
    {
        arr3[i]=arr1[i]+arr2[i];
        cout<<arr3[i]<<endl;
    }
}
