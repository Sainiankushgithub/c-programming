//  ** STACK WITH ALL OF ITS OPERATIONS
#include<bits/stdc++.h>
using namespace std;
class stack
{
    private:
    int top;
    int arr[5];
    public:
    stack()
    {
        top=-1;
        for(int i=0;i<5;i++)
        {
            arr[i]=0;
        }
    }
        bool isEmpty()
        {
            if(top==-1)
            {
                return true;
            }
            else{
                return false;
            }
        }
        bool isFull()
        {
            if(top==4)
            {
                return true;
            }
            else{
                return false;
            }
        }
        void push(int val)
        {
            if(isFull())
            {
                cout<<"Stack is overflow";
            }
            else{
                top++;
                arr[top]=val;
            }
        }
        int pop()
        {
            if(isEmpty())
            {
                cout<<"Stack is underflow";
            }
        }
        else{
            int popvalue=arr[top];
            arr[top]=0;
            top--;
            return popvalue;
        }
        int count()
        {
            return (top+1);
        }
        int peek(int pos)
        {
            if(isEmpty())
            {
                cout<<"Stack is underfow";
                return 0;
            }
            else{
                return arr[pos];
            }
        }
        int change(int pos, int val)
        {
            arr[pos]=val;
            cout<<"Value change at location "<<pos<<" is "<<val<<endl;
        }
        void display()
        {
            for(int i=4;i>=0;i--)
            {
                cout<<arr[i]<<endl;
            }
        }
    
};
